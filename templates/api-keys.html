import requests API_KEY = "lc_YOUR_KEY" BASE = "http://localhost:7860" AUTH = {"Authorization": f"Bearer {API_KEY}"} ## ── /v1/chat (JSON response) ────────────────────────────── # Text only r = requests.post(f"{BASE}/v1/chat", headers=AUTH, json={"message": "What is the capital of France?"}) print(r.json()["message"]["content"]) # With PDF or image file with open("report.pdf", "rb") as f: r = requests.post(f"{BASE}/v1/chat", headers=AUTH, data={"message": "Summarize this"}, files={"file": f}) print(r.json()["message"]["content"]) ## ── /api/ask (SSE streaming, no auth) ──────────────────── # Text only r = requests.post(f"{BASE}/api/ask", stream=True, json={"question": "Explain quantum physics simply"}) for line in r.iter_lines(): if line: import json d = json.loads(line.decode().replace("data: ", "")) if "token" in d: print(d["token"], end="", flush=True) if d.get("done"): break # With PDF or image file with open("report.pdf", "rb") as f: r = requests.post(f"{BASE}/api/ask", stream=True, data={"question": "Summarize this document"}, files={"file": f}) for line in r.iter_lines(): if line: d = json.loads(line.decode().replace("data: ", "")) if "token" in d: print(d["token"], end="", flush=True) if d.get("done"): break
